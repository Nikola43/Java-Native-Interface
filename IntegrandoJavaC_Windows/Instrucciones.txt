3. //Creamos fichero java
/* HelloWorld.java */

public class HelloWorld {
    native void helloFromC(); /* (1) */
    static {
        System.loadLibrary("ctest"); /* (2) */
    }
    static public void main(String argv[]) {
        HelloWorld helloWorld = new HelloWorld();
        helloWorld.helloFromC(); /* (3) */
    }
}

4. //Compilamos el fichero java
javac HelloWorld.java

5. //Generamos el fichero.h 
javah HelloWorld

6. //Creamos el fichero de c
/* ctest.c */

#include <jni.h>
#include <stdio.h>

JNIEXPORT void JNICALL Java_HelloWorld_helloFromC
  (JNIEnv * env, jobject jobj)
{
    printf("Hello from C!\n");
}

7. //Compilamos la libreria para 64 bits | usar -m32 para 32 bits

gcc ctest.c -o ctest.dll -m64 -shared -I "C:\Program Files\Java\jdk1.8.0_111\include" -I "C:\Program Files\Java\jdk1.8.0_111\include\win32"

8. //Ejecutar java
java HelloWorld
